
import streamlit as st
import google.generativeai as genai
import pandas as pd
import matplotlib.pyplot as plt

# ============================
#   CONFIGURE GEMINI API
# ============================
API_KEY = "#YOUR API KEY"
genai.configure(api_key=API_KEY)

model = genai.GenerativeModel("gemini-2.5-flash")

# ============================
#   STREAMLIT UI
# ============================
st.title("ðŸ¤– Chatbot Dashboard (Gemini AI)")
st.text("Welcome to your chatbot dashboard!")
st.header("Chat With Your Model")

# Input for user message
user_msg = st.text_input("Enter your message :")

# Button to send
if st.button("Send Message"):
    if user_msg.strip() != "":
        with st.spinner("Generating response..."):
            response = model.generate_content(user_msg)
        st.success("Bot Response:")
        st.write(response.text)

# File upload (Optional)
st.header("Upload a File (Optional)")
